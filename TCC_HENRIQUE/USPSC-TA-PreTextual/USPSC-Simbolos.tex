\begin{simbolos}
  \item[$\phi$] Porosidade do meio (poroso)
  \item[$K$] Permeabilidade absoluta (escalar ou tensor)
  \item[$\rho$]  Massa específica (densidade) do fluido
  \item[$\mu$] Viscosidade dinâmica do fluido
  \item[$p$] Pressão do fluido
  \item[$u$]  Velocidade de Darcy
  \item[$n$]  Vetor normal unitário
  \item[$c$] Concentração de um contaminante/traçador
  \item[$q$] Termo fonte/sumidouro (vazão por volume)
  \item[$h$] Passo espacial (tamanho da célula da malha)
  \item[$\Delta t$]  Passo temporal
  \item[$c_t$] Compressibilidade total
  \item[$c_r$] Compressibilidade da rocha
  \item[$c_f$] Compressibilidade do fluido
  \item[$\Omega$]  Domínio de estudo
  \item[$\partial \Omega$] Contorno (fronteira) do domínio
  \item[$O(\cdot)$]  Ordem de grandeza (notação Big O)
  \item[$\nu$] Número de Courant (CFL)
\end{simbolos}
